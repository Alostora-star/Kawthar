async function initializeDB() {
  try {
    let data = await AppData.findOne({ docId: "main_database" });
    if (!data) {
      data = new AppData();
      await data.save();
      console.log("ℹ️ تم إنشاء مستند البيانات الافتراضي بنجاح في السحاب لجامع الكوثر.");

      let adminUpdated = false;
      data.users = data.users.map(u => {
        if (u.id === "u-admin" || u.username === "admin") {
          u.password = "Shadow@123";
          adminUpdated = true;
        }
        return u;
